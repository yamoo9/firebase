# 데이터 업데이트

Firebase 데이터베이스의 데이터를 수정하려면 데이터베이스 참조 아이템에 update() 메서드를 사용합니다.

chaptersRef.child('key').update({
  description: '수정 내용'
});

# 에디트 모드





 
 
 








 
 



 
 
 
 



<template>
  <!-- ... -->
  <li v-for="chapter in chapters" :key="chapter.id">
    <!-- ... -->
    <p v-if="chapter === editingChapter">{{ chapter.description }}</p>
    <textarea v-else v-model="editingDesc"></textarea>
    <button type="button" @click="chageEditMode(chapter)">수정</button>
  </li>
</template>

<script>
export default {
  data () {
    return {
      // ...
      editingChapter: null,
      editingDesc: ''
    }
  },
  methods: {
  chageEditMode(chapter) {
    this.editingChapter = chapter;
    this.editDesc = chapter.description;
  }
}
</script>

# 에디트 모드의 버튼 표시

에디트 모드로 변경되면 취소, 업데이트 버튼이 화면에 표시되도록 설정합니다.

 



 
 
 
 

<div v-if="chapter !== editingChapter" class="dispay-mode">
  <button type="button" @click="removeChapter(chapter)">제거</button>
  <button type="button" @click="editChapter(chapter)">수정</button>
</div>
<div v-else class="edit-mode">
  <button type="button" @click="cancelEditing">취소</button>
  <button type="button" @click="updateChapter">업데이트</button>
</div>

# 에디트 모드의 메서드

에디트 모드에서 표시된 각 버튼에 연결된 취소, 업데이트 메서드를 정의합니다.

methods: {
  // ...
  cancelEditing() {
    this.editingChapter = null;
    this.editDesc = '';
  },
  updateChapter() {
    chaptersRef.child(this.editingChapter.id).update({
      description: this.editDesc
    });
    this.cancelEditing();
  }
}

# 업데이트 이벤트 감지

child_changed 이벤트는 데이터베이스 참조 아이템이 변경되면 이를 감지합니다. 컴포넌트 생성 시점에 child_changed 이벤트를 설정하여 참조 아이템이 변경되면 사용자 화면을 업데이트 하도록 코드를 작성합니다.

created() {
  // ...
  chaptersRef.on('child_changed', snapshot => {
    const updatedChapter = this.chapters.find(chapter => chapter.id === snapshot.key);
    updatedChapter.description = snapshot.val().description;
  });
}
