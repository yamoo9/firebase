# Native Toast 활용

native-toast 라이브러리를 활용해 Firebase의 데이터베이스 변경 사항을 사용자 화면에 알리도록 구현해봅니다.

# 설치

native-toast 모듈을 프로젝트에 설치합니다.

$ yarn add native-toast --dev  # 또는  npm i native-toast -D
1

# 사용법

사용법을 참고하여 컴포넌트 생성 시점에 Firebase 데이터베이스 아이템 추가, 변경, 제거 이벤트 구문에 각각 추가합니다.

 





 
 
 
 
 
 



 
 
 
 
 
 



 
 
 
 
 
 







import nativeToast from 'native-toast';

export default {
  created() {
    chaptersRef.on('child_added', snapshot => {
      // ...
      nativeToast({
        type: 'success',
        position: 'north-east',
        message: `${snapshot.val().title} 생성`,
        timeout: 3000,
        closeOnClick: true
      });
    });
    chaptersRef.on('child_changed', snapshot => {
      // ...
      nativeToast({
        type: 'info',
        position: 'north-east',
        message: `${snapshot.val().title} 내용 변경`,
        timeout: 3000,
        closeOnClick: true
      });
    });
    chaptersRef.on('child_removed', snapshot => {
      // ...
      nativeToast({
        type: 'warning',
        position: 'north-east',
        message: `${snapshot.val().title} 제거`,
        timeout: 3000,
        closeOnClick: true
      });
    });
  }
}
