# 이메일/패스워드 업데이트

# 이메일 업데이트

인증 사용자의 이메일 정보를 업데이트할 때는 updateEmail 메소드를 사용합니다.





 




const authUser = firebase.auth().currentUser;

if (authUser) {
  const newEmailAddress = 'new-type@email.com';
  authUser.updateEmail(newEmailAddress)
    .then(() => console.log('이메일 업데이트 성공!'))
    .catch(error => console.error('이메일 업데이트 실패!'));
}

# 패스워드 업데이트

인증 사용자의 패스워드 정보를 업데이트할 때는 updatePassword 메소드를 사용합니다.





 




const authUser = firebase.auth().currentUser;

if (authUser) {
  const newPassword = 'qwerty!23';
  authUser.updatePassword(newPassword)
    .then(() => console.log('패스워드 업데이트 성공!'))
    .catch(error => console.error('패스워드 업데이트 실패!'));
}
