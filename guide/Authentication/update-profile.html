# 프로필 업데이트

사용자의 표시 이름(displayName) 및 프로필 사진 URL(photoURL) 등 기본 프로필 정보를 업데이트할 때는 updateProfile 메소드를 사용합니다.









 
 
 


const authUser = firebase.auth().currentUser;

const updateProfileInfo = {
  displayName: '야무(yamoo9)',
  photoURL: 'https://goo.gl/Mi1oax'
};

if (authUser) {
  authUser.updateProfile(updateProfileInfo)
    .then(() => console.log('프로필 업데이트 성공!'))
    .catch(error => console.error('프로필 업데이트 실패!'));
}

Vue.js 애플리케이션 코드에서 프로필 업데이트를 반영하면 먼저 onAuthStateChanged 메서드 내에서 인증된 사용자가 전달되면 displayName, photoURL 속성 값을 컴포넌트 데이터 속성에 할당합니다.






 
 




 
 
 
 




import { auth } from 'firebase';

export default {
  data: () => ({
    authUser: null,
    displayName: '',
    photoURL: ''
  }),
  created() {
    auth().onAuthStateChanged(user => {
      this.authUser = user;
      if (user) {
        this.displayName = user.displayName;
        this.photoURL = user.photoURL;
      }
    });
  }
}

NOTE

이미지 업로드는 Firebase 스토리지(Storage)를 사용합니다.

uploadPhotoURL 메서드 코드를 작성합니다. 이미지 업로드 코드는 Firebase 스토리지에 파일 업로드를 공부한 후 살펴보세요.

uploadPhotoURL(e) {
  // 이미지 파일이 업로드 되면
  const file = e.target.files[0];
  const { name, size, type } = file;
  const metadata = { name, size, contentType: type };
  const extIndex = name.lastIndexOf('.');
  const ext = name.slice(extIndex);
  const filename = `${this.authUser.uid}-updated-photoURL${ext}`;
  // 파일을 Firebase 스토리지에 업로드
  storage().ref('/images').child(filename).put(file, metadata)
    .then(snapshot =>
      // Firebase 스토리지에서 downloadURL 값을 받아온 후
      snapshot.ref.getDownloadURL()
        // 컴포넌트의 photoURL 값으로 사용
        .then(url => (this.photoURL = url))
    );
},

updateAuthUserProfile 메서드 코드를 작성합니다. 인증된 사용자를 auth를 통해 참조한 후, 인증된 사용자가 있다면 displayName, photoURL 값을 컴포넌트 데이터로부터 가져와 인증 사용자의 updateProfile 메서드를 통해 업데이트 합니다.

updateAuthUserProfile() {
  // 사용자 계정의 updateProfile 메서드를 사용해
  const authUser = auth().currentUser;
  if (authUser) {
    const { displayName, photoURL } = this;
    // displayName, photoURL 업데이트
    authUser.updateProfile({ displayName, photoURL })
      .then(() => console.log('프로필 업데이트에 성공했습니다.'))
  }
}
