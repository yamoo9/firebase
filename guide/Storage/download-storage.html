# 파일 다운로드

Firebase 스토리지 파일을 다운로드 하려면 스토리지에서 이미지 파일을 참조한 후, getDownloadURL() 메서드를 사용한 다음 then() 문을 통해 downloadURL 값을 받아와 다운로드 코드를 작성합니다.






 
 
 












import { storage } from 'firebase';

export default {
  methods: {
    downloadAvatarImg() {
      storage().ref('images/avatar.jpg')
        .getDownloadURL()
        .then(url => {
          const link = document.createElement('a');
          link.setAttribute('href', url);
          link.setAttribute('download', 'avatar.jpg');
          document.body.appendChild(link);
          link.click();
          link.parentNode.removeChild(link);
        }
      );
    },
  }
}
